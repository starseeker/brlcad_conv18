# Include directories needed by libmultispectral users

	${BRLCAD_BINARY_DIR}/include
	${BRLCAD_SOURCE_DIR}/include

	${OPTICAL_INCLUDE_DIRS}
	)

    -DRT_MULTISPECTRAL
)
