		OUTPUT ${CMAKE_BINARY_DIR}/bin/archer

		DEPENDS ${CMAKE_CURRENT_SOURCE_DIR}/archer
		)
	add_custom_target(archer ALL DEPENDS ${CMAKE_BINARY_DIR}/bin/archer)
