# Minimum required version of CMake
CMAKE_MINIMUM_REQUIRED(VERSION 2.8)
IF(COMMAND CMAKE_POLICY)
  CMAKE_POLICY(SET CMP0003 NEW)
ENDIF(COMMAND CMAKE_POLICY)

# Set CMake project name
PROJECT(LEMON)

add_executable(lemon lemon.c)

if(NOT CMAKE_CONFIGURATION_TYPES)
  if(NOT TARGET_RUNTIME_OUTPUT_DIRECTORY)
    get_target_property(TARGET_RUNTIME_OUTPUT_DIRECTORY lemon RUNTIME_OUTPUT_DIRECTORY)
  endif(NOT TARGET_RUNTIME_OUTPUT_DIRECTORY)
  configure_file(lempar.c ${TARGET_RUNTIME_OUTPUT_DIRECTORY}/lempar.c COPY_ONLY)
else(NOT CMAKE_CONFIGURATION_TYPES)
  foreach(CFG_TYPE ${CMAKE_CONFIGURATION_TYPES})
	 string(TOUPPER "${CFG_TYPE}" CFG_TYPE_UPPER)
         if(NOT TARGET_RUNTIME_OUTPUT_DIRECTORY)
           get_target_property(TARGET_RUNTIME_OUTPUT_DIRECTORY lemon RUNTIME_OUTPUT_DIRECTORY_${CFG_TYPE_UPPER})
    	 endif(NOT TARGET_RUNTIME_OUTPUT_DIRECTORY)
	 configure_file(lempar.c ${TARGET_RUNTIME_OUTPUT_DIRECTORY}/lempar.c COPY_ONLY)
  endforeach(CFG_TYPE ${CMAKE_CONFIGURATION_TYPES})
endif(NOT CMAKE_CONFIGURATION_TYPES)
