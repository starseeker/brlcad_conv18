IF(HAVE_TERMLIB)
	SET(TERMLIB_INCLUDE_DIRS
		${BRLCAD_BINARY_DIR}/include
		${BRLCAD_SOURCE_DIR}/include
		${TERMLIB_INCLUDE_DIR}
		)
	LIST(REMOVE_DUPLICATES TERMLIB_INCLUDE_DIRS)
	SET(TERMLIB_INCLUDE_DIRS ${TERMLIB_INCLUDE_DIRS} CACHE STRING "Include directories for libtermlib" FORCE)
	MARK_AS_ADVANCED(TERMLIB_INCLUDE_DIRS)
	include_directories(${TERMLIB_INCLUDE_DIRS})
	BRLCAD_ADDLIB(libcursor cursor.c "${TERMLIB_LIBRARY}")
	SET_TARGET_PROPERTIES(libcursor PROPERTIES VERSION 19.0.1 SOVERSION 19)
	ADD_MAN_PAGE(3 libcursor.3)
	CMAKEFILES(Makefile.am)
ENDIF(HAVE_TERMLIB)
